set Serveroutput On;
Declare 
 TYPE NT_VAR_TYPE IS TABLE OF T_LIPO_POLICY%ROWTYPE;
 VAR NT_VAR_TYPE:=NT_VAR_TYPE();
 VAR2 T_LIPO_POLICY%ROWTYPE;
BEGIN
FOR I IN (SELECT * FROM T_LIPO_POLICY WHERE PO_PLN_CODE='723' AND PO_STATUS='K')
LOOP
  VAR2.PO_CONT:=I.Po_Cont;
  VAR2.PO_STATUS:=I.Po_Status;
  VAR2.PO_CO_ID:=I.Po_Co_Id;
  
  VAR.EXTEND();
  VAR(VAR.LAST):=VAR2;
  END LOOP;
Dbms_Output.Put_Line(Var.count);
END;
